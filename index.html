x_coords = np.linspace(-200, 200, 600)
y_coords = np.linspace(-200, 200, 600)
heart_x = []
heart_y = []
for y in y_coords:
    for x in x_coords:
        if ((x*0.01)**2+(y*0.01)**2-0.05)**3-(x*0.01)**2*(y*0.01)**3 <= 0:
            heart_x.append(x)
            heart_y.append(y)

plt.figure(figsize = FIG_SIZE)
plt.scatter(heart_x, heart_y, c="red")
plt.axis('off')
plt.savefig('heart.jpeg')

words_dic = {
    'kitten': 10,
    'cotton': 7,
    'love': 8,
    'Beijing': 5,
    'San_Diego': 3,
    'UCSD': 3,
    'miss_you': 5,
    'cute': 5,
    'time_location': 3,
    'eat_lotus_root': 1,
    'mua': 1,
    'voikunsama': 3,
    'minecx': 2,
}

words_lst = []
for w in words_dic:
    for i in range(words_dic[w]):
        words_lst.append(w)
        
for i in range(10):
    random.shuffle(words_lst)

words = ', '.join(words_lst)
